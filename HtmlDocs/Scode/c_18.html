 59  void TFT_Text_PrintChar( PT_T  *pos, char  ch )
 60  {
 61             uint8_t   glyph_index_entry;
 62      const  uint8_t  *glyph_data;                                /*  < start of the data >                         */
 63             PT_T      end_pos;
 64             COLOR_T  *pixel_color;
 65             uint8_t   bitmap_byte;
 66             uint8_t   glyph_width,
 67                       x_bm;
 68             uint32_t  offset;
 69             uint32_t  row,
 70                       col,
 71                       num_pixels;
 72  
 73      glyph_index_entry = ch - font_header->FirstChar;
 74      glyph_width       = glyph_index[ glyph_index_entry ].Width;
 75      offset            = glyph_index[ glyph_index_entry ].Offset;
 76      glyph_data        = & ( font[ offset ] );
 77  
 78      end_pos.X         = pos->X + CHAR_WIDTH - 1;
 79      end_pos.Y         = pos->Y + CHAR_HEIGHT - 1;
 80      TFT_Start_Rectangle( pos, &end_pos );
 81  
 82      for( row = 0; row < CHAR_HEIGHT; row++ )
 83      {
 84          x_bm = 0;
 85          do
 86          {
 87              bitmap_byte = *glyph_data;
 88              if( bitmap_byte == 0x00 )
 89              {
 90                  num_pixels = MIN( 8, glyph_width - x_bm );
 91                  TFT_Write_Rectangle_Pixel( &bg, num_pixels );
 92                  x_bm += num_pixels;
 93              }
 94              else
 95                  if( bitmap_byte == 0xff )
 96                  {
 97                      num_pixels = MIN( 8, glyph_width - x_bm );
 98                      TFT_Write_Rectangle_Pixel( &fg, num_pixels );
 99                      x_bm += num_pixels;
100                  }
101                  else
102                  {
103                      col        = 0;
104                      num_pixels = 0;
105                      if( ( bitmap_byte & 0x7f ) == 0 )
106                      {
107                          num_pixels = MIN( 7, glyph_width - x_bm );
108                          TFT_Write_Rectangle_Pixel( &bg, num_pixels );
109                      }
110                      else
111                          if( ( bitmap_byte & 0x7f ) == 0x7f )
112                          {
113                              num_pixels = MIN( 7, glyph_width - x_bm );
114                              TFT_Write_Rectangle_Pixel( &fg, num_pixels );
115                          }
116                          else
117                              if( ( bitmap_byte & 0x3f ) == 0 )
118                              {
119                                  num_pixels = MIN( 6, glyph_width - x_bm );
120                                  TFT_Write_Rectangle_Pixel( &bg, num_pixels );
121                              }
122                              else
123                                  if( ( bitmap_byte & 0x3f ) == 0x3f )
124                                  {
125                                      num_pixels = MIN( 6, glyph_width - x_bm );
126                                      TFT_Write_Rectangle_Pixel( &fg, num_pixels );
127                                  }
128                                  else
129                                      if( ( bitmap_byte & 0x1f ) == 0 )
130                                      {
131                                          num_pixels = MIN( 5, glyph_width - x_bm );
132                                          TFT_Write_Rectangle_Pixel( &bg, num_pixels );
133                                      }
134                                      else
135                                          if( ( bitmap_byte & 0x1f ) == 0x1f )
136                                          {
137                                              num_pixels = MIN( 5, glyph_width - x_bm );
138                                              TFT_Write_Rectangle_Pixel( &fg, num_pixels );
139                                          }
140                                          else
141                                              if( ( bitmap_byte & 0x0f ) == 0 )
142                                              {
143                                              num_pixels = MIN( 4, glyph_width - x_bm );
144                                              TFT_Write_Rectangle_Pixel( &bg, num_pixels );
145                                              }
146                                              else
147                                              if( ( bitmap_byte & 0x0f ) == 0x0f )
148                                              {
149                                              num_pixels = MIN( 4, glyph_width - x_bm );
150                                              TFT_Write_Rectangle_Pixel( &fg, num_pixels );
151                                              }
152                      if( num_pixels > 0 )
153                      {
154                          x_bm         += num_pixels;
155                          col          += num_pixels;
156                          bitmap_byte >>= num_pixels;
157                      }
158                      for( ; col < 8; col++ )
159                      {
160                          if( bitmap_byte & 0x01 )                /*             if pixel is to be set              */
161                              pixel_color = &fg;
162                          else
163                              pixel_color = &bg;
164                          TFT_Write_Rectangle_Pixel( pixel_color, 1 );
165                          bitmap_byte >>= 1;
166                          x_bm++;
167                          if( x_bm >= glyph_width )
168                              break;
169                      }
170                  }
171              glyph_data++;
172          } while( x_bm < glyph_width );
173          if( x_bm < CHAR_WIDTH )
174          {
175          // fill in rest of cell with background color for narrow glyphs
176              TFT_Write_Rectangle_Pixel( &bg, CHAR_WIDTH - x_bm );
177          }
178      }
179  }

181  void TFT_Text_PrintStr( PT_T  *pos, char  *str )
182  {
183      while( *str )
184      {
185          TFT_Text_PrintChar( pos, *str );
186          str++;
187          pos->X += CHAR_WIDTH;                                   /*     forces monospacing for fonts               */
188      }
189  }
